@app.route('/login', methods=['POST'])
def login():
    username = request.form.get('username')
    password = request.form.get('password')

    conn = get_connection()
    cursor = conn.cursor(dictionary=True)
    cursor.execute("SELECT * FROM `user` WHERE user_name=%s AND user_password=%s", (username, password))
    user = cursor.fetchone()
    cursor.close()
    conn.close()

    if user:
        # Save user session
        session['user_id'] = user['user_id']
        session['user_name'] = user['user_name']
        session['personal_name'] = user['personal_name']
        session['job_desc'] = user['job_desc']
        return jsonify({"success": True})

    return jsonify({"success": False})

@app.route('/activity_Tables')
def activity_Tables():
    if 'user_id' not in session:
        return redirect(url_for('login_page'))

    user_info = {
        "user_name": session.get('user_name'),
        "personal_name": session.get('personal_name'),
        "job_desc": session.get('job_desc')
    }

    conn = get_connection()
    cursor = conn.cursor(dictionary=True)

    query = """
        SELECT 
            t.Table_db_id,
            t.Table_number,
            tf.floor_name AS table_floor,

            COALESCE(tr.table_status, 'available') AS table_status,
            tr.Date_reserved,
            tr.number_of_people

        FROM tables t

        LEFT JOIN table_floor tf
            ON t.Table_db_id = tf.Table_db_id

        LEFT JOIN (
            SELECT r1.*
            FROM table_reservations r1
            INNER JOIN (
                SELECT Table_db_id, MAX(Date_reserved) AS max_date
                FROM table_reservations
                GROUP BY Table_db_id
            ) r2
            ON r1.Table_db_id = r2.Table_db_id
            AND r1.Date_reserved = r2.max_date
        ) tr
            ON t.Table_db_id = tr.Table_db_id

        ORDER BY tf.floor_name, t.Table_number
    """

    cursor.execute(query)
    all_tables = cursor.fetchall()

    cursor.close()
    conn.close()

    return render_template(
        'activity_Tables.html',
        user=user_info,
        tables=all_tables
    )

@app.route('/place_order_table', methods=['POST'])
def place_order_table():
    if 'user_id' not in session:
        return jsonify({"success": False, "error": "User not logged in"})

    data = request.get_json()
    order_number = data.get('order_number')
    order_type = data.get('order_type')  # dine_in / takeaway
    customer_name = data.get('customer_name')
    table_id = data.get('table_id')  # Table_db_id from dropdown
    selected_products = data.get('selected_products', [])  # array of good names

    if not order_number or not order_type or not customer_name:
        return jsonify({"success": False, "error": "Missing required fields"})
    if not selected_products:
        return jsonify({"success": False, "error": "No products selected"})

    try:
        conn = get_connection()
        cursor = conn.cursor()

        # 1️⃣ Insert into order_table
        cursor.execute("""
            INSERT INTO order_table (order_number, Dateandtime_ordered)
            VALUES (%s, %s)
        """, (order_number, datetime.now()))
        order_db_id = cursor.lastrowid

        # 2️⃣ Insert into order_types
        cursor.execute("""
            INSERT INTO order_types (order_db_id, order_type)
            VALUES (%s, %s)
        """, (order_db_id, order_type))

        # 3️⃣ Insert into customer_orders_verification
        cursor.execute("""
            INSERT INTO customer_orders_verification 
            (customer_name_order, order_db_id, user_id, order_table_location)
            VALUES (%s, %s, %s, %s)
        """, (customer_name, order_db_id, session['user_id'], table_id))

        # Get the newly inserted customer_order_id
        cursor.execute("""
            SELECT customer_order_id 
            FROM customer_orders_verification 
            WHERE order_db_id=%s
        """, (order_db_id,))
        customer_order_id = cursor.fetchone()[0]

        # 4️⃣ Insert selected goods into customer_goods_purchased
        for good_name in selected_products:
            # Get good_number from products table
            cursor.execute("SELECT good_number FROM products WHERE good_name=%s", (good_name,))
            row = cursor.fetchone()
            if row:
                good_number = row[0]
                cursor.execute("""
                    INSERT INTO customer_goods_purchased (customer_order_id, good_number)
                    VALUES (%s, %s)
                """, (customer_order_id, good_number))

        conn.commit()
        cursor.close()
        conn.close()

        return jsonify({"success": True})

    except Error as e:
        print("Database Error:", e)
        return jsonify({"success": False, "error": str(e)})

                    item.addEventListener('click', function(e) {
                        e.stopPropagation();
                        const dropdownBtn = this.closest('.dropdown').querySelector('.dropdown-btn');
                        dropdownBtn.textContent = this.textContent + ' ▼';
                        dropdownBtn.dataset.selectedId = this.dataset.id;

                        this.parentElement.querySelectorAll('.dot').forEach(dot => dot.style.background = 'transparent');
                        this.querySelector('.dot').style.background = 'lightblue';
                    });

                    item.addEventListener('dblclick', function(e) {
                        e.stopPropagation();
                        const dropdownBtn = this.closest('.dropdown').querySelector('.dropdown-btn');
                        dropdownBtn.textContent = dropdownBtn.dataset.defaultText + ' ▼';
                        delete dropdownBtn.dataset.selectedId;
                        this.querySelector('.dot').style.background = 'transparent';
                    });
                });
            }

            document.querySelectorAll('.dropdown-content').forEach(content => {
                setupDropdownItems(content.querySelectorAll('.dropdown-item'));
            });

            document.addEventListener('click', function() {
                document.querySelectorAll('.dropdown-content').forEach(content => {
                    content.style.display = 'none';
                });
            });
        });

                    customerName.focus();

                    customerName.contentEditable = "false";
                    orderNumber.contentEditable = "false";
                    customerName.style.borderBottom = "none";

            currentQty++;

        // SELECT PRODUCT
        selectBtn.addEventListener('click', () => {
            if (!selectedProduct) return;

            const priceNum = Number(selectedProduct.price); // force number
            const totalPrice = (priceNum * currentQty).toFixed(2);

            const itemRow = document.createElement('div');
            itemRow.style.display = 'flex';
            itemRow.style.alignItems = 'center';
            itemRow.style.gap = '10px';
            itemRow.style.marginBottom = '8px';

            itemRow.innerHTML = `
                <img src="${selectedProduct.imageSrc}" style="width:40px; height:40px; object-fit:cover; border-radius:6px;">
                <div style="flex:1;">
                    <div style="font-weight:bold;">${selectedProduct.name}</div>
                    <div style="color:green;">$${totalPrice}</div>
                </div>
                <div>Qty: ${currentQty}</div>
            `;

            orderItems.appendChild(itemRow);

            tempOrderItems.push({
                name: selectedProduct.name,
                price: priceNum,
                quantity: currentQty,
                sub_total: priceNum * currentQty, // numeric value
                imageSrc: selectedProduct.imageSrc
            });

            productModal.style.display = 'none';

            updateSubtotal();
            updateTaxDisplay();
            updateTotal();

        function updateSubtotal() {
            const subtotalEl = document.querySelector('.billing-row span.amount'); // first .amount span
            const subtotal = tempOrderItems.reduce((acc, item) => acc + item.sub_total, 0);
            subtotalEl.textContent = `$${subtotal.toFixed(2)}`;
        }

        function calculateTax(subtotal) {
                const taxRate = 0.1; // 10%
                return subtotal * taxRate;
        }

        function updateTaxDisplay() {
            const subtotal = tempOrderItems.reduce((acc, item) => acc + item.sub_total, 0);
            const tax = calculateTax(subtotal);

            // Ensure tax is always a number and formatted to 2 decimals
            document.getElementById('taxAmount').textContent = `$${Number(tax.toFixed(2))}`;
        }

            // PROMO-CODE BUTTON FUNCTIONALITY
        const promoBtn = document.querySelector('.pill-btn_order'); // assumes first button is Promo-code
        const discountDisplay = document.getElementById('discountDisplay');

        promoBtn.addEventListener('click', () => {
            let discountPercent = prompt("Enter discount percentage (e.g., 10 for 10%)");
            
            if (discountPercent !== null) {
                discountPercent = Number(discountPercent);
                
                if (isNaN(discountPercent) || discountPercent < 0 || discountPercent > 100) {
                    alert("Please enter a valid number between 0 and 100");
                    return;
                }

                // Show the discount percentage
                discountDisplay.textContent = discountPercent + "%";

                // Store discount for future use (optional)
                tempOrderItems.discountPercent = discountPercent;
            }
        });

        function updateTotal() {
            const subtotal = tempOrderItems.reduce((acc, item) => acc + item.sub_total, 0);

            const tax = calculateTax(subtotal);

            const discountPercent = Number(tempOrderItems.discountPercent || 0);
            const discountAmount = subtotal * (discountPercent / 100);

            const total = subtotal + tax - discountAmount;

            document.getElementById('totalAmount').textContent = `$${total.toFixed(2)}`;
        }

        const tableBtn = document.querySelector('.dropdown-btn[data-default-text="Select Table"]');

        placeOrderBtn.addEventListener('click', () => {
            const orderNumber = orderNumberDiv.textContent.trim();
            const orderType = orderTypeBtn.dataset.selectedId; // dine_in or takeaway
            const customerName = document.getElementById('customerName').textContent.trim();
            const tableBtn = document.querySelector('.dropdown-btn'); // your table dropdown button
            const tableId = tableBtn.dataset.selectedId;

            // Get selected products from your actual display screen
            const orderItemsDivs = document.querySelectorAll('#orderItems div');
            const selectedProducts = Array.from(orderItemsDivs).map(d => d.textContent.trim());

            if (!orderNumber || orderNumber === 'Order Number') {
                alert('Please enter Order Number');
                return;
            }
            if (!orderType) {
                alert('Please select Order Type');
                return;
            }
            if (!customerName || customerName === 'Customer Name') {
                alert('Please enter Customer Name');
                return;
            }
            if (!tableId) {
                alert('Please select a table');
                return;
            }
            if (selectedProducts.length === 0) {
                alert('Please select at least one product');
                return;
            }

            fetch('/place_order_table', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    order_number: orderNumber,
                    order_type: orderType,
                    customer_name: customerName,
                    table_id: tableId,
                    selected_products: selectedProducts
                })
            })
            .then(res => res.json())
            .then(data => {
                if (data.success) {
                    alert('Order placed successfully ✅');
                } else {
                    alert('Failed to place order ❌: ' + data.error);
                }
            })
            .catch(err => {
                console.error(err);
                alert('Server error ❌');
            });
        });
